#   - seq handling
#   - command/response
#   - sustained streaming telemetry
#
# This test is intentionally sensor-free and should be repeatable on any bench setup.

# Protocol
# ---------------------------
frame_format:      [AA][55][TYPE][SEQ][LEN][PAYLOAD...][CRC16 lo][CRC16 hi]
crc16_variant:     CRC-16/CCITT-FALSE (poly=0x1021, init=0xFFFF, refin=false, refout=false, xorout=0x0000)
crc_input:         [TYPE][SEQ][LEN][PAYLOAD...]

# Hardware under test

# ---------------------------
# Test procedure
# ---------------------------
# 1) Flash protocol_compliance.ino to MCU.
# 2) On host, run MATLAB test:
#       pipeline_test_with_commands(<PORT>, 2000000)
#    Expected:
#       PING seq=0 -> PONG OK
#       SET_STREAM_US(5000) seq=1 -> ACK status=0
#       STREAM_ENABLE(1) seq=2 -> ACK status=0
#       Stream of SENSE_FRAME at ~200 Hz for 10 seconds.
#
# Notes:
# - CMD_SET_STREAM_US sets telemetry period in microseconds.
# - SENSE_FRAME payload length is 36 bytes (synthetic pattern in compliance sketch).

# ---------------------------
# Expected outputs / pass criteria
# ---------------------------
pass_criteria:
  - PONG received matching seq for CMD_PING
  - ACKNOWLEDGED received matching seq for CMD_SET_STREAM_US with status=0
  - ACKNOWLEDGED received matching seq for CMD_STREAM_ENABLE with status=0
  - During 10 s window: SENSE_FRAME count >= 1800 (>= 180 Hz average)  # conservative
  - No persistent CRC failures (occasional loss acceptable on USB, but should be rare)
